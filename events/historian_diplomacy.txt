namespace = historian_diplomacy


# 触发记录国际评价的事件
country_event = {
	id = historian_diplomacy.1
	hide_window = yes

	trigger = { is_ai = no }
	is_triggered_only = yes
	fire_only_once = yes

	after = {
		country_event = {
			id = historian_diplomacy.2
			days = 9
		}
	}
}

# 记录我国与其它国家相互的评价
country_event = {
	id = historian_diplomacy.2
	hide_window = yes

	trigger = { is_ai = no }
	is_triggered_only = yes

	immediate = {
		every_country = {
			limit = { 
				has_communications = Root
				or = {
					is_country_type = default
					is_country_type = fallen_empire
					is_country_type = awakened_fallen_empire
				}
			}

			# 我方的评价
			export_trigger_value_to_variable = {
				trigger = their_opinion
				parameters = { who = Root }
				variable = HIS_OUR_OPINION
			}
			# 对方的评价
			export_trigger_value_to_variable = {
				trigger = opinion
				parameters = { who = Root }
				variable = HIS_OPINION_ON_US
			}

			log = "HIS_OPINION:[GetDate],[This.GetName],[This.HIS_OUR_OPINION],[This.HIS_OPINION_ON_US]"
		}
	}
	after = {
		country_event = {
			id = historian_diplomacy.2
			days = 360
		}
	}
}
